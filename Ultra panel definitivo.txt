-- Función para crear secciones
local function createSection(name, parent)
    local Section = Instance.new("Frame")
    Section.Size = UDim2.new(1, -20, 0, 160)
    Section.Position = UDim2.new(0, 10, 0, (#parent:GetChildren()-1)*170)
    Section.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    Section.BorderSizePixel = 0
    Section.Parent = parent

    local Corner = Instance.new("UICorner", Section)
    Corner.CornerRadius = UDim.new(0, 10)

    local SectionTitle = Instance.new("TextLabel", Section)
    SectionTitle.Size = UDim2.new(1, 0, 0, 35)
    SectionTitle.BackgroundTransparency = 1
    SectionTitle.Text = name
    SectionTitle.TextScaled = true
    SectionTitle.Font = Enum.Font.FredokaOne
    SectionTitle.TextColor3 = Color3.fromRGB(0, 255, 255)

    return Section
end

-- Función para crear botones toggle estilizados
local function createToggle(parent, text, callback)
    local Button = Instance.new("TextButton")
    Button.Size = UDim2.new(1, -20, 0, 40)
    Button.Position = UDim2.new(0, 10, 0, (#parent:GetChildren()-1)*45 + 35)
    Button.BackgroundColor3 = Color3.fromRGB(0, 180, 255)
    Button.TextColor3 = Color3.fromRGB(0, 0, 0)
    Button.Font = Enum.Font.FredokaOne
    Button.TextScaled = true
    Button.Text = text.." [OFF]"
    Button.Parent = parent

    local Corner = Instance.new("UICorner", Button)
    Corner.CornerRadius = UDim.new(0, 8)

    local Gradient = Instance.new("UIGradient", Button)
    Gradient.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(0, 255, 255)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(0, 150, 255))
    }

    -- Hover effect
    Button.MouseEnter:Connect(function()
        Button.BackgroundColor3 = Color3.fromRGB(0, 220, 255)
    end)
    Button.MouseLeave:Connect(function()
        Button.BackgroundColor3 = Color3.fromRGB(0, 180, 255)
    end)

    local toggled = false
    Button.MouseButton1Click:Connect(function()
        toggled = not toggled
        Button.Text = text.." ["..(toggled and "ON" or "OFF").."]"
        callback(toggled)
    end)
end
